SELECT pub_id, pub_titulo, SUBSTRING(pub_descripcion, 1, 100) AS pub_descripcion, pub_precio, pub_id_categoria, pub_id_usuario, pub_id_tipo_publicacion, pub_imagen

FROM publicaciones

WHERE pub_id_categoria = 13 

AND pub_titulo LIKE "%lenovo%"

AND `pub_precio` >= 4000

AND `pub_precio` <= 10000

ORDER BY pub_precio DESC
